http.Response.Body未被完全读取的问题

August 9, 2024 · 563 words · 2 min · Http

经常在一些开源的包中(例如：dghubble/sling)处理http响应体的时候出现这么一段代码：defer io.Copy(io.Discard, resp.Body)。这段代码的作用主要是丢弃掉响应体中的内容。为什么要这么做呢？

经过查询http库源代码后看到一段注释.

// Body represents the response body.
//
// The response body is streamed on demand as the Body field
// is read. If the network connection fails or the server
// terminates the response, Body.Read calls return an error.
//
// The http Client and Transport guarantee that Body is always
// non-nil, even on responses without a body or responses with
// a zero-length body. It is the caller's responsibility to
// close Body. The default HTTP client's Transport may not
// reuse HTTP/1.x "keep-alive" TCP connections if the Body is
// not read to completion and closed.
//
// The Body is automatically dechunked if the server replied
// with a "chunked" Transfer-Encoding.
//
// As of Go 1.12, the Body will also implement io.Writer
// on a successful "101 Switching Protocols" response,
// as used by WebSockets and HTTP/2's "h2c" mode.
Body io.ReadCloser

下面可以通过一段代码验证这个观点：

package main  
  
import (  
    "fmt"  
    "io"
	"log"    
	"net"    
	"net/http"    
	"net/http/httptrace"    
	"time"
	)  
  
func main() {  
    err := do()  
    if err != nil {  
       log.Fatal(err)  
    }}  
  
func do() error {  
    client := &http.Client{  
       Timeout: 5 * time.Second,  
       Transport: &http.Transport{  
          Proxy: http.ProxyFromEnvironment,  
          DialContext: (&net.Dialer{  
             Timeout: 2 * time.Second,  
          }).DialContext,  
          ForceAttemptHTTP2:     false,  
          MaxIdleConns:          100,  
          MaxIdleConnsPerHost:   10,  
          IdleConnTimeout:       10 * time.Second,  
          TLSHandshakeTimeout:   2 * time.Second,  
          ResponseHeaderTimeout: 2 * time.Second,  
       },    
    }    
    trace := &httptrace.ClientTrace{  
	   GotConn: func(connInfo httptrace.GotConnInfo) {  
		  fmt.Printf("Got Conn: %+v\n", connInfo)  
	   },    
    }  
    fmt.Printf("multiGetReadBody:\n")  
    if err := nultiGetReadBody(client, trace); err != nil {  
       return err  
    }  
    client.CloseIdleConnections()  
  
    fmt.Printf("\nmultiGetNotReadBody:\n")  
    if err := nultiGetNotReadBody(client, trace); err != nil {  
       return err  
    }  
    return nil  
}  
  
func nultiGetReadBody(client *http.Client, trace *httptrace.ClientTrace) error {  
    if err := get(client, trace, true); err != nil {  
       return err  
    }  
    if err := get(client, trace, true); err != nil {  
       return err  
    }  
    if err := get(client, trace, true); err != nil {  
       return err  
    }  
    return nil  
}  
  
func nultiGetNotReadBody(client *http.Client, trace *httptrace.ClientTrace) error {  
    if err := get(client, trace, false); err != nil {  
       return err  
    }  
    if err := get(client, trace, false); err != nil {  
       return err  
    }  
    if err := get(client, trace, false); err != nil {  
       return err  
    }  
    return nil  
}  
  
func get(client *http.Client, trace *httptrace.ClientTrace, readBody bool) error {  
    req, err := http.NewRequest("GET", "http://baidu.com/", nil)  
    if err != nil {  
       return err  
    }  
    req = req.WithContext(httptrace.WithClientTrace(req.Context(), trace))  
    resp, err := client.Do(req)  
    if err != nil {  
       return err  
    }  
    defer func() {  
       if err := resp.Body.Close(); err != nil {  
          fmt.Println("close err:", err)  
       }    }()    if readBody {  
       _, err := io.Copy(io.Discard, resp.Body)  
       if err != nil {  
          return nil  
       }  
    }    return nil  
}
